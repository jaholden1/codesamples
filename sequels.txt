CREATE TABLE "films" (
    "id" INTEGER PRIMARY KEY AUTOINCREMENT NOT NULL,
    "title" TEXT NOT NULL,
    "year" INTEGER NOT NULL
);
CREATE TABLE "sequels" (
    "id" INTEGER PRIMARY KEY AUTOINCREMENT NOT NULL,
    "film_id" INTEGER NOT NULL,  
    "title" TEXT NOT NULL,
    "year" INTEGER NOT NULL,
    CONSTRAINT FK_film FOREIGN KEY (film_id)
    REFERENCES films(id)
);
CREATE TABLE "increase" (
    "film_id" INTEGER NOT NULL,
    "sequel_id" INTEGER NOT NULL,
    "increase" NUMERIC NOT NULL,
    CONSTRAINT FK_film FOREIGN KEY (film_id)
    REFERENCES films(id)
    CONSTRAINT FK_sequel FOREIGN KEY (sequel_id)
    REFERENCES sequels(id)
);
