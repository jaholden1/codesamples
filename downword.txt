def downword(word):
    # convert to lowercase 
    lword = word.lower();
    # iterate through length of string
    for i in range(len(lword) - 1):
        # if current char is less than char to the right, it is not a downword
        if lword[i] < lword[i + 1]:
            return False
    return True
